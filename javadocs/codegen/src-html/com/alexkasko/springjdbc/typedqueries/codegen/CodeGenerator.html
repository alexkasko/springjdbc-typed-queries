040        /**
041         * Constructor
042         *
043         *
044         * @param isPublic whether generated class and its methods will have 'public' access modifier
045         * @param useIterableJdbcTemplate whether to use iterable jdbc template extensions from this
046         *                                project (https://github.com/alexkasko/springjdbc-iterable)
047         * @param useCheckSingleRowUpdates whether to generate additional update methods, those check that
048    *                                 only single row was changed on update
049         * @param useBatchInserts whether to generate additional insert (DML) methods (with parameters), those
050    *                        takes {@link java.util.Iterator} of parameters and execute inserts
051    *                        for the contents of the specified iterator in batch mode
052         * @param useTemplateStringSubstitution whether to recognize query templates on method generation
053         * @param selectRegex regular expression to use for identifying 'select' queries by name
054         * @param updateRegex regular expression to use for identifying 'insert', 'update' and 'delete' queries by name
055         * @param templateRegex regular expression to recognize query templates by name
056         * @param templateValueConstraintRegex regular expression constraint for template substitution values
057         * @param typeIdMap mapping of parameter names postfixes to data types
058         * @param freemarkerTemplate freemarker template body
059         * @param freemarkerConf freemarker configuration    @throws CodeGeneratorException on any error
060         */
061        public CodeGenerator(boolean isPublic, boolean useIterableJdbcTemplate, boolean useCheckSingleRowUpdates,
062                             boolean useBatchInserts, boolean useTemplateStringSubstitution, String selectRegex,
063                             String updateRegex, String templateRegex, String templateValueConstraintRegex,
064                             Map<String, Class<?>> typeIdMap, String freemarkerTemplate,
065                             Configuration freemarkerConf) throws CodeGeneratorException {
066            if(null == selectRegex) throw new CodeGeneratorException("Provided selectRegex is null");
067            if(null == updateRegex) throw new CodeGeneratorException("Provided updateRegex is null");
068            if(null == templateRegex) throw new CodeGeneratorException("Provided templateRegex is null");
069            if(null == templateValueConstraintRegex) throw new CodeGeneratorException("Provided templateValueConstraintRegex is null");
070            if(null == typeIdMap) throw new CodeGeneratorException("Provided typeIdMap is null");
071            if(null == freemarkerTemplate) throw new CodeGeneratorException("Provided freemarkerTemplate is null");
072            if(null == freemarkerConf) throw new CodeGeneratorException("Provided freemarkerConf is null");
073            this.isPublic = isPublic;
074            this.useIterableJdbcTemplate = useIterableJdbcTemplate;
075            this.useCheckSingleRowUpdates = useCheckSingleRowUpdates;
076            this.useBatchInserts = useBatchInserts;;
077            this.useTemplateStringSubstitution = useTemplateStringSubstitution;
078            this.selectRegex = Pattern.compile(selectRegex);
079            this.updateRegex = Pattern.compile(updateRegex);
080            this.templateRegex = Pattern.compile(templateRegex);
081            this.templateValueConstraintRegex = Pattern.compile(templateValueConstraintRegex);
082            this.typeIdMap = typeIdMap;
083            this.freemarkerTemplate = freemarkerTemplate;
084            this.freemarkerConf = freemarkerConf;
085        }

087        /**
088         * Generates queries wrappers class
089         *
090         * @param queries name -> text query mapping
091         * @param fullClassName class name with package prefix
092         * @param sourceSqlFileName name of source SQL file
093         * @param output output writer
094         * @throws CodeGeneratorException
095         */
096        public void generate(Map<String, String> queries, String fullClassName, String sourceSqlFileName, Writer output)
097                throws CodeGeneratorException {
098            if(null == queries || 0 == queries.size()) throw new CodeGeneratorException("Provided queries map is empty");
099            if(null == fullClassName || 0 == fullClassName.length()) throw new CodeGeneratorException("Provided fullClassName is empty");
100            if(!fullClassName.contains(".") || (fullClassName.lastIndexOf(".") ==  fullClassName.length())) throw new CodeGeneratorException(
101                    "Full class name must have non empty package prefix, but was: [" + fullClassName + "]");
102            if(null == output) throw new CodeGeneratorException("Provided output is null");
103            try {
104                RootTemplateArg params = createTemplateArgs(queries, fullClassName, sourceSqlFileName);
105                Reader templateReader = new StringReader(freemarkerTemplate);
106                Template ftl = new Template(fullClassName, templateReader, freemarkerConf, "UTF-8");
107                ftl.process(params, output);
108            } catch (TemplateException e) {
109                throw new CodeGeneratorException("Error processing FreeMarker template", e);
110            } catch (IOException e) {
111                throw new CodeGeneratorException("IO error", e);
112            } catch (Exception e) {
113                if(e instanceof CodeGeneratorException) throw (CodeGeneratorException) e;
114                throw new CodeGeneratorException("Error generating queries code", e);
115            }
116        }
117    
118        private RootTemplateArg createTemplateArgs(Map<String, String> queries, String fullClassName, String sourceSqlFileName) {
119            int dotIndex = fullClassName.lastIndexOf(".");
120            String packageName = fullClassName.substring(0, dotIndex);
121            String className = fullClassName.substring(dotIndex + 1);
122            String modifier = isPublic ? "public" : "";
123            List<QueryTemplateArg> selects = new ArrayList<QueryTemplateArg>();
124            List<QueryTemplateArg> updates = new ArrayList<QueryTemplateArg>();
125            for (Map.Entry<String, String> en : queries.entrySet()) {
126                String name = en.getKey();
127                String sql = en.getValue();
128                List<ParamTemplateArg> params = createNamedParams(sql);
129                boolean isTemplate = useTemplateStringSubstitution && templateRegex.matcher(name).matches();
130                QueryTemplateArg query = new QueryTemplateArg(name, params, isTemplate);
131                if (selectRegex.matcher(name).matches()) selects.add(query);
132                else if (updateRegex.matcher(name).matches()) updates.add(query);
133                else throw new CodeGeneratorException("Invalid query name: [" + name + "], names must match select regex: " +
134                            "[" + selectRegex + "] or updateRegex: [" + updateRegex + "]");
135            }
136            return new RootTemplateArg(packageName, className, modifier, useIterableJdbcTemplate, useCheckSingleRowUpdates,
137                    useBatchInserts, useTemplateStringSubstitution, sourceSqlFileName, templateValueConstraintRegex.pattern(),
138                    selects, updates);
139        }
140    
141        private List<ParamTemplateArg> createNamedParams(String sql) {
142            List<String> paramNames = parseParamsNames(sql);
143            List<ParamTemplateArg> args = new ArrayList<ParamTemplateArg>(paramNames.size());
144            for (String name : paramNames) {
145                args.add(new ParamTemplateArg(name, typeForName(name)));
146            }
147            return args;
148        }
149    
150        private Class typeForName(String name) {
151            for (Map.Entry<String, Class<?>> en : typeIdMap.entrySet()) {
152                String postfix = en.getKey();
153                if (name.endsWith(postfix)) return en.getValue();
154            }
155            return Object.class;
156        }

170            sb.append('}');
171            return sb.toString();
172        }

451                Configuration fc = new Configuration();

453                fc.setTagSyntax(SQUARE_BRACKET_TAG_SYNTAX);

461            private static String defaultFreemarkerTemplate() {
462                InputStream is = null;
463                try {
464                    is = CodeGenerator.class.getResourceAsStream("/com/alexkasko/springjdbc/typedqueries/codegen/BeanQueries.ftl");
465                    byte[] data = readFull(is);
466                    return new String(data, Charset.forName("UTF-8"));
467                } catch (IOException e) {
468                    throw new RuntimeException(e);
469                } finally {
470                    closeQuietly(is);
471                }
472            }
473    
474            private static byte[] readFull(InputStream is) throws IOException {
475                ByteArrayOutputStream baos = new ByteArrayOutputStream();
476                byte[] buffer = new byte[1024 * 4];
477                int n;
478                while (-1 != (n = is.read(buffer))) {
479                    baos.write(buffer, 0, n);
480                }
481                return baos.toByteArray();
482            }
483    
484            private static void closeQuietly(InputStream is) {
485                try {
486                    if (is != null) {
487                        is.close();
488                    }
489                } catch (IOException ioe) {
490                    // ignore
491                }
492            }
