001    package com.alexkasko.springjdbc.typedqueries.codegen;
002    
003    import java.util.Collection;
004    
005    /**
006     * Query argument class for freemarker template
007     * Made public to conform freemarker's requirements
008     *
009     * @author alexkasko
010     * Date: 12/23/12
011     */
012    public class QueryTemplateArg {
013        private final String name;
014        private final Collection<ParamTemplateArg> params;
015    
016        /**
017         * Constructor
018         *
019         * @param name query name
020         * @param params query parameters list
021         */
022        QueryTemplateArg(String name, Collection<ParamTemplateArg> params) {
023            this.name = name;
024            this.params = params;
025        }
026    
027        /**
028         * Query name accessor
029         *
030         * @return query name
031         */
032        public String getName() {
033            return name;
034        }
035    
036        /**
037         * Query parameters list accessor
038         *
039         * @return query parameters list
040         */
041        public Collection<ParamTemplateArg> getParams() {
042            return params;
043        }
044    
045        /**
046         * {@inheritDoc}
047         */
048        @Override
049        public String toString() {
050            final StringBuilder sb = new StringBuilder();
051            sb.append("QueryTemplateArg");
052            sb.append("{name='").append(name).append('\'');
053            sb.append(", params=").append(params);
054            sb.append('}');
055            return sb.toString();
056        }
057    }
