001package com.alexkasko.springjdbc.typedqueries.codegen;
002
003import java.util.Collection;
004
005/**
006 * Query argument class for freemarker template
007 * Made public to conform freemarker's requirements
008 *
009 * @author alexkasko
010 * Date: 12/23/12
011 */
012public class QueryTemplateArg {
013    private final String name;
014    private final Collection<ParamTemplateArg> params;
015    private final Collection<ParamTemplateArg> columns;
016    private final boolean template;
017
018    /**
019     * Constructor
020     *
021     * @param name query name
022     * @param params query parameters list
023     * @param columns
024     * @param template whether this query needs template substitutions
025     */
026    QueryTemplateArg(String name, Collection<ParamTemplateArg> params,
027                     Collection<ParamTemplateArg> columns, boolean template) {
028        this.name = name;
029        this.params = params;
030        this.columns = columns;
031        this.template = template;
032    }
033
034    /**
035     * Query name accessor
036     *
037     * @return query name
038     */
039    public String getName() {
040        return name;
041    }
042
043    /**
044     * Query parameters list accessor
045     *
046     * @return query parameters list
047     */
048    public Collection<ParamTemplateArg> getParams() {
049        return params;
050    }
051
052    /**
053     * Select-clause column aliases accessor
054     *
055     * @return select-clause column aliases accessor
056     */
057    public Collection<ParamTemplateArg> getColumns() {
058        return columns;
059    }
060
061    /**
062     * Whether this query needs template substitutions
063     *
064     * @return whether this query needs template substitutions
065     */
066    public boolean isTemplate() { return template; }
067
068    /**
069     * {@inheritDoc}
070     */
071    @Override
072    public String toString() {
073        final StringBuilder sb = new StringBuilder();
074        sb.append("QueryTemplateArg");
075        sb.append("{name='").append(name).append('\'');
076        sb.append(", params=").append(params);
077        sb.append(", columns=").append(columns);
078        sb.append(", template=").append(template);
079        sb.append('}');
080        return sb.toString();
081    }
082}
