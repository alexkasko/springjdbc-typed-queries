001package com.alexkasko.springjdbc.typedqueries.codegen;
002
003import java.text.SimpleDateFormat;
004import java.util.Collection;
005import java.util.Date;
006
007/**
008 * Root argument class for freemarker template.
009 * Made public to conform freemarker's requirements
010 *
011 * @author alexkasko
012 * Date: 12/22/12
013 */
014public class RootTemplateArg {
015    private static final String DATE_FORMAT = "yyyy-MM-dd HH:mm:ss";
016    private final String packageName;
017    private final String className;
018    private final String modifier;
019    private final boolean useIterableJdbcTemplate;
020    private final boolean useCloseableIterables;
021    private final boolean useCheckSingleRowUpdates;
022    private final boolean useBatchInserts;
023    private final boolean useTemplateStringSubstitution;
024    private final boolean useUnderscoredToCamel;
025    private final boolean generateInterfacesForColumns;
026    private final boolean useFluentSettersForColumns;
027    private final String sourceSqlFileName;
028    private final String templateValueConstraintRegex;
029    private final Collection<QueryTemplateArg> selects;
030    private final Collection<QueryTemplateArg> updates;
031
032    /**
033     * Constructor
034     *
035     * @param packageName package name for generated class
036     * @param className generated class name
037     * @param modifier class and methods modifier, may be 'public', empty (package-private) by default
038     * @param useIterableJdbcTemplate whether to use JdbcTemplate extension from this project
039*                                (https://github.com/alexkasko/springjdbc-iterable)
040     * @param useCloseableIterables whether to generate {@code CloseableIterable} methods
041     * @param useCheckSingleRowUpdates whether to generate additional update methods, those check that
042*                                 only single row was changed on update
043     * @param useBatchInserts whether to generate additional insert (DML) methods (with parameters), those
044*                        takes {@link java.util.Iterator} of parameters and execute inserts
045*                        for the contents of the specified iterator in batch mode
046     * @param useTemplateStringSubstitution whether to recognize query templates on method generation
047     * @param useUnderscoredToCamel whether to convert underscored parameter named to camel ones
048     * @param generateInterfacesForColumns whether to generate interfaces for columns
049     * @param useFluentSettersForColumns whether to use fluent setters for columns
050     * @param sourceSqlFileName name of source SQL file
051     * @param templateValueConstraintRegex regular expression constraint for template substitution values
052     * @param selects list of 'select' queries
053     * @param updates list of 'update' queries
054     */
055    RootTemplateArg(String packageName, String className, String modifier, boolean useIterableJdbcTemplate,
056                    boolean useCloseableIterables, boolean useCheckSingleRowUpdates, boolean useBatchInserts, boolean useTemplateStringSubstitution, boolean useUnderscoredToCamel, boolean generateInterfacesForColumns, boolean useFluentSettersForColumns, String sourceSqlFileName,
057                    String templateValueConstraintRegex, Collection<QueryTemplateArg> selects, Collection<QueryTemplateArg> updates) {
058        this.packageName = packageName;
059        this.className = className;
060        this.modifier = modifier;
061        this.useIterableJdbcTemplate = useIterableJdbcTemplate;
062        this.useCloseableIterables = useCloseableIterables;
063        this.useCheckSingleRowUpdates = useCheckSingleRowUpdates;
064        this.useBatchInserts = useBatchInserts;
065        this.useTemplateStringSubstitution = useTemplateStringSubstitution;
066        this.useUnderscoredToCamel = useUnderscoredToCamel;
067        this.generateInterfacesForColumns = generateInterfacesForColumns;
068        this.useFluentSettersForColumns = useFluentSettersForColumns;
069        this.sourceSqlFileName = sourceSqlFileName;
070        this.templateValueConstraintRegex = templateValueConstraintRegex;
071        this.selects = selects;
072        this.updates = updates;
073    }
074
075    /**
076     * Package name accessor
077     *
078     * @return package name
079     */
080    public String getPackageName() {
081        return packageName;
082    }
083
084    /**
085     * Class name accessor
086     *
087     * @return class name
088     */
089    public String getClassName() {
090        return className;
091    }
092
093    /**
094     * Modifier accessor
095     *
096     * @return modifier
097     */
098    public String getModifier() {
099        return modifier.length() > 0 ? modifier + " " : modifier;
100    }
101
102    /**
103     * Whether to use iterable extensions flag accessor
104     *
105     * @return whether to use iterable extensions flag
106     */
107    public boolean isUseIterableJdbcTemplate() {
108        return useIterableJdbcTemplate;
109    }
110
111    /**
112     * Whether to generate {@code CloseableIterable} methods
113     *
114     * @return whether to generate {@code CloseableIterable} methods
115     */
116    public boolean isUseCloseableIterables() {
117        return useCloseableIterables;
118    }
119
120    /**
121     * Whether to generate additional update methods, those check that
122     * only single row was changed on update
123     *
124     * @return whether to generate update-check methods
125     */
126    public boolean isUseCheckSingleRowUpdates() { return useCheckSingleRowUpdates; }
127
128    /**
129     * Whether to generate additional insert (DML) methods (with parameters), those
130     * takes {@link java.util.Iterator} of parameters and execute inserts
131     * for the contents of the specified iterator in batch mode
132     *
133     * @return whether to generate batch-insert methods
134     */
135    public boolean isUseBatchInserts() { return useBatchInserts; }
136
137    /**
138     * Whether to recognize query templates on method generation
139     *
140     * @return whether to recognize query templates on method generation
141     */
142    public boolean isUseTemplateStringSubstitution() { return useTemplateStringSubstitution; }
143
144    /**
145     * Whether to convert underscored parameter named to camel ones
146     *
147     * @return whether to convert underscored parameter named to camel ones
148     */
149    public boolean isUseUnderscoredToCamel() {
150        return useUnderscoredToCamel;
151    }
152
153    /**
154     * Whether to generate interfaces for columns
155     *
156     * @return whether to generate interfaces for columns
157     */
158    public boolean isGenerateInterfacesForColumns() {
159        return generateInterfacesForColumns;
160    }
161
162    /**
163     * Whether to use fluent setters for columns
164     *
165     * @return whether to use fluent setters for columns
166     */
167    public boolean isUseFluentSettersForColumns() {
168        return useFluentSettersForColumns;
169    }
170
171    /**
172     * Source SQL file name accessor
173     *
174     * @return source SQL file name
175     */
176    public String getSourceSqlFileName() {
177        return sourceSqlFileName;
178    }
179
180    /**
181     * Regular expression constraint for template substitution values
182     *
183     * @return regular expression constraint for template substitution values
184     */
185    public String getTemplateValueConstraintRegex() {
186        return templateValueConstraintRegex;
187    }
188
189    /**
190     * 'Select' queries accessor
191     *
192     * @return 'select' queries list
193     */
194    public Collection<QueryTemplateArg> getSelects() {
195        return selects;
196    }
197
198    /**
199     * 'Update' queries accessor
200     *
201     * @return 'update' queries list
202     */
203    public Collection<QueryTemplateArg> getUpdates() {
204        return updates;
205    }
206
207    /**
208     * Current date accessor
209     *
210     * @return current date
211     */
212    public String getCurrentDate() {
213        return new SimpleDateFormat(DATE_FORMAT).format(new Date());
214    }
215
216    /**
217     * {@inheritDoc}
218     */
219    @Override
220    public String toString() {
221        final StringBuilder sb = new StringBuilder();
222        sb.append("RootTemplateArg");
223        sb.append("{packageName='").append(packageName).append('\'');
224        sb.append(", className='").append(className).append('\'');
225        sb.append(", modifier='").append(modifier).append('\'');
226        sb.append(", useIterableJdbcTemplate=").append(useIterableJdbcTemplate);
227        sb.append(", useCloseableIterables=").append(useCloseableIterables);
228        sb.append(", useCheckSingleRowUpdates=").append(useCheckSingleRowUpdates);
229        sb.append(", useBatchInserts=").append(useBatchInserts);
230        sb.append(", useTemplateStringSubstitution=").append(useTemplateStringSubstitution);
231        sb.append(", sourceSqlFileName='").append(sourceSqlFileName).append('\'');
232        sb.append(", templateValueConstraintRegex='").append(templateValueConstraintRegex).append('\'');
233        sb.append(", selects=").append(selects);
234        sb.append(", updates=").append(updates);
235        sb.append('}');
236        return sb.toString();
237    }
238}
