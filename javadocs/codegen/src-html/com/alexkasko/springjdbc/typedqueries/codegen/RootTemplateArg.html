001    package com.alexkasko.springjdbc.typedqueries.codegen;
002    
003    import java.text.SimpleDateFormat;
004    import java.util.Collection;
005    import java.util.Date;
006    
007    /**
008     * Root argument class for freemarker template.
009     * Made public to conform freemarker's requirements
010     *
011     * @author alexkasko
012     * Date: 12/22/12
013     */
014    public class RootTemplateArg {
015        private static final String DATE_FORMAT = "yyyy-MM-dd HH:mm:ss";
016        private final String packageName;
017        private final String className;
018        private final String modifier;
019        private final boolean useIterableJdbcTemplate;
020        private final boolean useCheckSingleRowUpdates;
021        private final boolean useBatchInserts;
022        private final boolean useTemplateStringSubstitution;
023        private final boolean useUnderscoredToCamel;
024        private final boolean generateInterfacesForColumns;
025        private final String sourceSqlFileName;
026        private final String templateValueConstraintRegex;
027        private final Collection<QueryTemplateArg> selects;
028        private final Collection<QueryTemplateArg> updates;
029    
030        /**
031         * Constructor
032         *
033         * @param packageName package name for generated class
034         * @param className generated class name
035         * @param modifier class and methods modifier, may be 'public', empty (package-private) by default
036         * @param useIterableJdbcTemplate whether to use JdbcTemplate extension from this project
037    *                                (https://github.com/alexkasko/springjdbc-iterable)
038         * @param useCheckSingleRowUpdates whether to generate additional update methods, those check that
039    *                                 only single row was changed on update
040         * @param useBatchInserts whether to generate additional insert (DML) methods (with parameters), those
041    *                        takes {@link java.util.Iterator} of parameters and execute inserts
042    *                        for the contents of the specified iterator in batch mode
043         * @param useTemplateStringSubstitution whether to recognize query templates on method generation
044         * @param useUnderscoredToCamel whether to convert underscored parameter named to camel ones
045         * @param generateInterfacesForColumns whether to generate interfaces for columns
046         * @param sourceSqlFileName name of source SQL file
047         * @param templateValueConstraintRegex regular expression constraint for template substitution values
048         * @param selects list of 'select' queries
049         * @param updates list of 'update' queries
050         */
051        RootTemplateArg(String packageName, String className, String modifier, boolean useIterableJdbcTemplate,
052                        boolean useCheckSingleRowUpdates, boolean useBatchInserts, boolean useTemplateStringSubstitution, boolean useUnderscoredToCamel, boolean generateInterfacesForColumns, String sourceSqlFileName,
053                        String templateValueConstraintRegex, Collection<QueryTemplateArg> selects, Collection<QueryTemplateArg> updates) {
054            this.packageName = packageName;
055            this.className = className;
056            this.modifier = modifier;
057            this.useIterableJdbcTemplate = useIterableJdbcTemplate;
058            this.useCheckSingleRowUpdates = useCheckSingleRowUpdates;
059            this.useBatchInserts = useBatchInserts;
060            this.useTemplateStringSubstitution = useTemplateStringSubstitution;
061            this.useUnderscoredToCamel = useUnderscoredToCamel;
062            this.generateInterfacesForColumns = generateInterfacesForColumns;
063            this.sourceSqlFileName = sourceSqlFileName;
064            this.templateValueConstraintRegex = templateValueConstraintRegex;
065            this.selects = selects;
066            this.updates = updates;
067        }
068    
069        /**
070         * Package name accessor
071         *
072         * @return package name
073         */
074        public String getPackageName() {
075            return packageName;
076        }
077    
078        /**
079         * Class name accessor
080         *
081         * @return class name
082         */
083        public String getClassName() {
084            return className;
085        }
086    
087        /**
088         * Modifier accessor
089         *
090         * @return modifier
091         */
092        public String getModifier() {
093            return modifier.length() > 0 ? modifier + " " : modifier;
094        }
095    
096        /**
097         * Whether to use iterable extensions flag accessor
098         *
099         * @return whether to use iterable extensions flag
100         */
101        public boolean isUseIterableJdbcTemplate() {
102            return useIterableJdbcTemplate;
103        }
104    
105        /**
106         * Whether to generate additional update methods, those check that
107         * only single row was changed on update
108         *
109         * @return whether to generate update-check methods
110         */
111        public boolean isUseCheckSingleRowUpdates() { return useCheckSingleRowUpdates; }
112    
113        /**
114         * Whether to generate additional insert (DML) methods (with parameters), those
115         * takes {@link java.util.Iterator} of parameters and execute inserts
116         * for the contents of the specified iterator in batch mode
117         *
118         * @return whether to generate batch-insert methods
119         */
120        public boolean isUseBatchInserts() { return useBatchInserts; }
121    
122        /**
123         * Whether to recognize query templates on method generation
124         *
125         * @return whether to recognize query templates on method generation
126         */
127        public boolean isUseTemplateStringSubstitution() { return useTemplateStringSubstitution; }
128    
129        /**
130         * Whether to convert underscored parameter named to camel ones
131         *
132         * @return whether to convert underscored parameter named to camel ones
133         */
134        public boolean isUseUnderscoredToCamel() {
135            return useUnderscoredToCamel;
136        }
137    
138        /**
139         * Whether to generate interfaces for columns
140         *
141         * @return whether to generate interfaces for columns
142         */
143        public boolean isGenerateInterfacesForColumns() {
144            return generateInterfacesForColumns;
145        }
146    
147        /**
148         * Source SQL file name accessor
149         *
150         * @return source SQL file name
151         */
152        public String getSourceSqlFileName() {
153            return sourceSqlFileName;
154        }
155    
156        /**
157         * Regular expression constraint for template substitution values
158         *
159         * @return regular expression constraint for template substitution values
160         */
161        public String getTemplateValueConstraintRegex() {
162            return templateValueConstraintRegex;
163        }
164    
165        /**
166         * 'Select' queries accessor
167         *
168         * @return 'select' queries list
169         */
170        public Collection<QueryTemplateArg> getSelects() {
171            return selects;
172        }
173    
174        /**
175         * 'Update' queries accessor
176         *
177         * @return 'update' queries list
178         */
179        public Collection<QueryTemplateArg> getUpdates() {
180            return updates;
181        }
182    
183        /**
184         * Current date accessor
185         *
186         * @return current date
187         */
188        public String getCurrentDate() {
189            return new SimpleDateFormat(DATE_FORMAT).format(new Date());
190        }
191    
192        /**
193         * {@inheritDoc}
194         */
195        @Override
196        public String toString() {
197            final StringBuilder sb = new StringBuilder();
198            sb.append("RootTemplateArg");
199            sb.append("{packageName='").append(packageName).append('\'');
200            sb.append(", className='").append(className).append('\'');
201            sb.append(", modifier='").append(modifier).append('\'');
202            sb.append(", useIterableJdbcTemplate=").append(useIterableJdbcTemplate);
203            sb.append(", useCheckSingleRowUpdates=").append(useCheckSingleRowUpdates);
204            sb.append(", useBatchInserts=").append(useBatchInserts);
205            sb.append(", useTemplateStringSubstitution=").append(useTemplateStringSubstitution);
206            sb.append(", sourceSqlFileName='").append(sourceSqlFileName).append('\'');
207            sb.append(", templateValueConstraintRegex='").append(templateValueConstraintRegex).append('\'');
208            sb.append(", selects=").append(selects);
209            sb.append(", updates=").append(updates);
210            sb.append('}');
211            return sb.toString();
212        }
213    }
