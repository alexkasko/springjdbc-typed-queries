001    package com.alexkasko.springjdbc.typedqueries.codegen;
002    
003    import java.text.SimpleDateFormat;
004    import java.util.Collection;
005    import java.util.Date;
006    
007    /**
008     * Root argument class for freemarker template.
009     * Made public to conform freemarker's requirements
010     *
011     * @author alexkasko
012     * Date: 12/22/12
013     */
014    public class RootTemplateArg {
015        private static final String DATE_FORMAT = "yyyy-MM-dd HH:mm:ss";
016        private final String packageName;
017        private final String className;
018        private final String modifier;
019        private final boolean useIterableJdbcTemplate;
020        private final String sourceSqlFileName;
021        private final Collection<QueryTemplateArg> selects;
022        private final Collection<QueryTemplateArg> updates;
023    
024        /**
025         * Constructor
026         *
027         * @param packageName package name for generated class
028         * @param className generated class name
029         * @param modifier class and methods modifier, may be 'public', empty (package-private) by default
030         * @param useIterableJdbcTemplate whether to use JdbcTemplate extension from this project
031         *                                (https://github.com/alexkasko/springjdbc-iterable)
032         * @param sourceSqlFileName name of source SQL file
033         * @param selects list of 'select' queries
034         * @param updates list of 'update' queries
035         */
036        RootTemplateArg(String packageName, String className, String modifier, boolean useIterableJdbcTemplate,
037                               String sourceSqlFileName, Collection<QueryTemplateArg> selects, Collection<QueryTemplateArg> updates) {
038            this.packageName = packageName;
039            this.className = className;
040            this.modifier = modifier;
041            this.useIterableJdbcTemplate = useIterableJdbcTemplate;
042            this.sourceSqlFileName = sourceSqlFileName;
043            this.selects = selects;
044            this.updates = updates;
045        }
046    
047        /**
048         * Package name accessor
049         *
050         * @return package name
051         */
052        public String getPackageName() {
053            return packageName;
054        }
055    
056        /**
057         * Class name accessor
058         *
059         * @return class name
060         */
061        public String getClassName() {
062            return className;
063        }
064    
065        /**
066         * Modifier accessor
067         *
068         * @return modifier
069         */
070        public String getModifier() {
071            return modifier.length() > 0 ? modifier + " " : modifier;
072        }
073    
074        /**
075         * Whether to use iterable extensions flag accessor
076         *
077         * @return whether to use iterable extensions flag
078         */
079        public boolean isUseIterableJdbcTemplate() {
080            return useIterableJdbcTemplate;
081        }
082    
083        /**
084         * Source SQL file name accessor
085         *
086         * @return source SQL file name
087         */
088        public String getSourceSqlFileName() {
089            return sourceSqlFileName;
090        }
091    
092        /**
093         * 'Select' queries accessor
094         *
095         * @return 'select' queries list
096         */
097        public Collection<QueryTemplateArg> getSelects() {
098            return selects;
099        }
100    
101        /**
102         * 'Update' queries accessor
103         *
104         * @return 'update' queries list
105         */
106        public Collection<QueryTemplateArg> getUpdates() {
107            return updates;
108        }
109    
110        /**
111         * Current date accessor
112         *
113         * @return current date
114         */
115        public String getCurrentDate() {
116            return new SimpleDateFormat(DATE_FORMAT).format(new Date());
117        }
118    
119        /**
120         * {@inheritDoc}
121         */
122        @Override
123        public String toString() {
124            final StringBuilder sb = new StringBuilder();
125            sb.append("RootTemplateArg");
126            sb.append("{packageName='").append(packageName).append('\'');
127            sb.append(", className='").append(className).append('\'');
128            sb.append(", modifier='").append(modifier).append('\'');
129            sb.append(", useIterableJdbcTemplate=").append(useIterableJdbcTemplate);
130            sb.append(", sourceSqlFileName='").append(sourceSqlFileName).append('\'');
131            sb.append(", selects=").append(selects);
132            sb.append(", updates=").append(updates);
133            sb.append('}');
134            return sb.toString();
135        }
136    }
