111    /**
112     * Generates queries wrappers class
113     *
114     * @param queries name -> text query mapping
115     * @param fullClassName class name with package prefix
116     * @param sourceSqlFileName name of source SQL file
117     * @param output output writer
118     * @throws CodeGeneratorException
119     */
120    public void generate(Map<String, String> queries, String fullClassName, String sourceSqlFileName, Writer output)
121            throws CodeGeneratorException {
122        if(null == queries || 0 == queries.size()) throw new CodeGeneratorException("Provided queries map is empty");
123        if(null == fullClassName || 0 == fullClassName.length()) throw new CodeGeneratorException("Provided fullClassName is empty");
124        if(!fullClassName.contains(".") || (fullClassName.lastIndexOf(".") ==  fullClassName.length())) throw new CodeGeneratorException(
125                "Full class name must have non empty package prefix, but was: [" + fullClassName + "]");
126        if(null == output) throw new CodeGeneratorException("Provided output is null");
127        try {
128            RootTemplateArg params = createTemplateArgs(queries, fullClassName, sourceSqlFileName);
129            Reader templateReader = new StringReader(freemarkerTemplate);
130            Template ftl = new Template(fullClassName, templateReader, freemarkerConf, "UTF-8");
131            ftl.process(params, output);
132        } catch (TemplateException e) {
133            throw new CodeGeneratorException("Error processing FreeMarker template", e);
134        } catch (IOException e) {
135            throw new CodeGeneratorException("IO error", e);
136        } catch (Exception e) {
137            if(e instanceof CodeGeneratorException) throw (CodeGeneratorException) e;
138            throw new CodeGeneratorException("Error generating queries code", e);
139        }
140    }

142    private RootTemplateArg createTemplateArgs(Map<String, String> queries, String fullClassName, String sourceSqlFileName) {
143        int dotIndex = fullClassName.lastIndexOf(".");
144        String packageName = fullClassName.substring(0, dotIndex);
145        String className = fullClassName.substring(dotIndex + 1);
146        String modifier = isPublic ? "public" : "";
147        List<QueryTemplateArg> selects = new ArrayList<QueryTemplateArg>();
148        List<QueryTemplateArg> updates = new ArrayList<QueryTemplateArg>();
149        for (Map.Entry<String, String> en : queries.entrySet()) {
150            String name = en.getKey();
151            String sql = en.getValue();
152            List<String> paramNames = parseParamsNames(sql);
153            Set<ParamTemplateArg> params = createNamedParams(paramNames);
154            List<String> colNames = parseColumnNames(sql);
155            Set<ParamTemplateArg> columns = createNamedParams(colNames);
156            boolean isTemplate = useTemplateStringSubstitution && templateRegex.matcher(name).matches();
157            QueryTemplateArg query = new QueryTemplateArg(name, params, columns, isTemplate);
158            if (selectRegex.matcher(name).matches()) selects.add(query);
159            else if (updateRegex.matcher(name).matches()) updates.add(query);
160            else throw new CodeGeneratorException("Invalid query name: [" + name + "], names must match select regex: " +
161                        "[" + selectRegex + "] or updateRegex: [" + updateRegex + "]");
162        }
163        return new RootTemplateArg(packageName, className, modifier, useIterableJdbcTemplate, useCloseableIterables, useCheckSingleRowUpdates,
164                useBatchInserts, useTemplateStringSubstitution, useUnderscoredToCamel, generateInterfacesForColumns, useFluentSettersForColumns, sourceSqlFileName, templateValueConstraintRegex.pattern(),
165                selects, updates);
166    }

168    private Set<ParamTemplateArg> createNamedParams(List<String> rawParamNames) {
169        final List<String> paramNames;
170        if(useUnderscoredToCamel) {
171            paramNames = new ArrayList<String>(rawParamNames.size());
172            for(String pa : rawParamNames) paramNames.add(underscoredToCamel(pa));
173        } else paramNames = rawParamNames;
174        Set<ParamTemplateArg> args = new LinkedHashSet<ParamTemplateArg>(paramNames.size());
175        for (String name : paramNames) {
176            args.add(new ParamTemplateArg(name, typeForName(name)));
177        }
178        return args;
179    }

181    private Class typeForName(String name) {
182        for (Map.Entry<String, Class<?>> en : typeIdMap.entrySet()) {
183            String postfix = en.getKey();
184            if (name.endsWith(postfix)) return en.getValue();
185        }
186        return Object.class;
187    }

190    static List<String> parseColumnNames(String sql) {
191        if(!sql.toLowerCase(ENGLISH).startsWith("select")) return emptyList();
192        // clean from parenthesed contents
193        int level = 0;
194        StringBuilder cleanSb = new StringBuilder();
195        for (int i = 0; i < sql.length(); i++) {
196            char ch = sql.charAt(i);
197            if ('(' == ch) {
198                if (0 == level) cleanSb.append('(');
199                level += 1;
200            } else if (')' == ch) {
201                level -= 1;
202                if (0 == level) cleanSb.append(')');
203            } else if (0 == level) cleanSb.append(ch);
204        }
205        // remove all after first from
206        String colstring = COLUMNS_FROM_REGEX.matcher(cleanSb.toString()).replaceFirst("");
207        // parse last word before comma
208        String[] selectParts = COLUMNS_COMMA_REGEX.split(colstring);
209        List<String> res = new ArrayList<String>(selectParts.length);
210        for(String pa : selectParts) {
211            String[] colParts = COLUMNS_SPACE_REGEX.split(pa);
212            String colname = colParts[colParts.length - 1];
213            if(COLUMNS_NAME_RESTRICTION_PATTERN.matcher(colname).matches()) res.add(colname);
214        }
215        return res;
216    }

218    static String underscoredToCamel(String underscored) {
219        if(null == underscored || 0 == underscored.length() || !underscored.contains("_")) return underscored;
220        StringBuilder sb = new StringBuilder();
221        boolean usFound = false;
222        for(int i = 0; i< underscored.length(); i++) {
223            char ch = underscored.charAt(i);
224            if('_' == ch) {
225                if(usFound) { // double underscore
226                    sb.append('_');
227                } else {
228                    usFound = true;
229                }
230            } else if (usFound) {
231                sb.append(toUpperCase(ch));
232                usFound = false;
233            } else {
234                sb.append(ch);
235            }
236        }
237        if(usFound) sb.append("_");
238        return sb.toString();
239    }

241    static String camelToUnderscored(String camel) {
242        if (null == camel || camel.length() < 2) return camel;
243        boolean hasUpper = false;
244        for (int i = 1; i < camel.length(); i++) {
245            char ch = camel.charAt(i);
246            if (ch == Character.toUpperCase(ch)) {
247                hasUpper = true;
248                break;
249            }
250        }
251        if (!hasUpper) return camel;
252        StringBuilder sb = new StringBuilder();
253        sb.append(camel.charAt(0));
254        for (int i = 1; i < camel.length(); i++) {
255            char ch = camel.charAt(i);
256            char lower = toLowerCase(ch);
257            if (ch == toUpperCase(ch) && ch != lower) {
258                sb.append("_");
259                sb.append(lower);
260            } else {
261                sb.append(ch);
262            }
263        }
264        return sb.toString();
265    }

632            ByteArrayOutputStream baos = new ByteArrayOutputStream();
633            byte[] buffer = new byte[1024 * 4];
634            int n;
635            while (-1 != (n = is.read(buffer))) {
