001    package com.alexkasko.springjdbc.typedqueries.codegen;
002    
003    /**
004     * Query parameter argument class for freemarker template
005     * Made public to conform freemarker's requirements
006     *
007     * @author alexkasko
008     * Date: 12/23/12
009     */
010    public class ParamTemplateArg {
011        private final String name;
012        private final Class<?> type;
013    
014        /**
015         * Constructor
016         *
017         * @param name parameter name
018         * @param type parameter type
019         */
020        ParamTemplateArg(String name, Class<?> type) {
021            this.name = name;
022            this.type = type;
023        }
024    
025        /**
026         * Parameter name accessor
027         *
028         * @return parameter name
029         */
030        public String getName() {
031            return name;
032        }
033    
034        /**
035         * Parameter type accessor
036         *
037         * @return parameter type
038         */
039        public String getType() {
040            return type.getSimpleName();
041        }
042    
043        /**
044         * {@inheritDoc}
045         */
046        @Override
047        public String toString() {
048            final StringBuilder sb = new StringBuilder();
049            sb.append("ParamTemplateArg");
050            sb.append("{name='").append(name).append('\'');
051            sb.append(", type=").append(type);
052            sb.append('}');
053            return sb.toString();
054        }
055    }
