001package com.alexkasko.springjdbc.typedqueries.common;
002
003import java.io.*;
004import java.nio.charset.Charset;
005import java.util.LinkedHashMap;
006import java.util.Map;
007import java.util.regex.Matcher;
008import java.util.regex.Pattern;
009
010/**
011 * Parses SQL queries file (as {@link Reader} or {@link InputStream}) to 'name'->'sql' map.
012 * Query name syntax: {@code /** selectSomeData STAR_SYMBOL/ }
013 * Won't close provided stream/reader.
014 *
015 * @author alexkasko
016 * Date: 12/22/12
017 */
018public class PlainSqlQueriesParser {
019    private enum State {STARTED, COLLECTING}
020    public static final Pattern DEFAULT_QUERY_NAME_REGEX = Pattern.compile("^\\s*/\\*{2}\\s*(.*?)\\s*\\*/\\s*$");
021
022    /**
023     * Parses provided stream to 'name'->'sql' map using 'UTF-8' encoding
024     *
025     * @param is sql queries file
026     * @return 'name'->'sql' map.
027     */
028    public Map<String, String> parse(InputStream is) {
029        return parse(is, "UTF-8");
030    }
031
032    /**
033     * Parses provided stream to 'name'->'sql' map.
034     *
035     * @param is sql queries file
036     * @param encoding sql queries file encoding
037     * @return 'name'->'sql' map.
038     */
039    public Map<String, String> parse(InputStream is, String encoding) {
040        return parse(new InputStreamReader(is, Charset.forName(encoding)));
041    }
042
043    /**
044     * Parses provided reader to 'name'->'sql' map.
045     *
046     * @param reader sql queries file
047     * @return 'name'->'sql' map.
048     */
049    public Map<String, String> parse(Reader reader) throws SqlFileParseException {
050        Map<String, String> res = new LinkedHashMap<String, String>();
051        BufferedReader re = null;
052        try {
053            re = new BufferedReader(reader);
054            State state = State.STARTED;
055            String name = null;
056            StringBuilder sql = new StringBuilder();
057            String line;
058            while(null != (line = re.readLine())) {
059                String trimmed = line.trim();
060                if(0 == trimmed.length()) continue;
061                switch(state) {
062                    case STARTED: // search for first query name
063                        if(trimmed.startsWith("--")) continue;
064                        Matcher startedMatcher = DEFAULT_QUERY_NAME_REGEX.matcher(line);
065                        if(!startedMatcher.matches()) throw new SqlFileParseException(
066                                "Query name not found on start, regex: [" + DEFAULT_QUERY_NAME_REGEX + "]");
067                        name = startedMatcher.group(1);
068                        state = State.COLLECTING;
069                        break;
070                    case COLLECTING:
071                        Matcher nameMatcher = DEFAULT_QUERY_NAME_REGEX.matcher(line);
072                        if(nameMatcher.matches()) { // next query name found
073                            if(0 == sql.length()) throw new SqlFileParseException(
074                                    "No SQL found for request name: [" + name + "]");
075                            // save collected sql string
076                            String existed = res.put(name, sql.toString());
077                            if(null != existed) throw new SqlFileParseException(
078                                    "Duplicate SQL query name: [" + name + "]");
079                            // clean collected sql string
080                            sql = new StringBuilder();
081                            name = nameMatcher.group(1);
082                        } else {
083                            sql.append(line);
084                            sql.append("\n");
085                        }
086                        break;
087                    default: throw new SqlFileParseException(state.name());
088                }
089            }
090            // tail
091            if(null == name) throw new SqlFileParseException("No queries found");
092            String existed = res.put(name, sql.toString());
093            if(null != existed) throw new SqlFileParseException("Duplicate SQL query name: [" + name + "]");
094        } catch (IOException e) {
095            throw new SqlFileParseException(e);
096        } catch (Exception e) {
097            if(e instanceof SqlFileParseException) throw (SqlFileParseException) e;
098            throw new SqlFileParseException(e);
099        } finally {
100            closeQuietly(re);
101        }
102        return res;
103    }
104
105    private static void closeQuietly(Closeable is) {
106        try {
107            if (is != null) {
108                is.close();
109            }
110        } catch (IOException ioe) {
111            // ignore
112        }
113    }
114}
