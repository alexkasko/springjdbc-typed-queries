001package com.alexkasko.springjdbc.typedqueries.common;
002
003import org.springframework.jdbc.core.RowMapper;
004import org.springframework.jdbc.core.SingleColumnRowMapper;
005
006import java.math.BigDecimal;
007import java.util.Date;
008
009/**
010 * Typed queries common utilities
011 *
012 * @author alexkasko
013 * Date: 2/5/13
014 */
015public class TypedQueriesUtils {
016    public static final RowMapper<String> STRING_ROW_MAPPER = new SingleColumnRowMapper<String>(String.class);
017    public static final RowMapper<Boolean> BOOLEAN_ROW_MAPPER = new SingleColumnRowMapper<Boolean>(Boolean.class);
018    public static final RowMapper<Short> SHORT_ROW_MAPPER = new SingleColumnRowMapper<Short>(Short.class);
019    public static final RowMapper<Integer> INT_ROW_MAPPER = new SingleColumnRowMapper<Integer>(Integer.class);
020    public static final RowMapper<Long> LONG_ROW_MAPPER = new SingleColumnRowMapper<Long>(Long.class);
021    public static final RowMapper<Float> FLOAT_ROW_MAPPER = new SingleColumnRowMapper<Float>(Float.class);
022    public static final RowMapper<Double> DOUBLE_ROW_MAPPER = new SingleColumnRowMapper<Double>(Double.class);
023    public static final RowMapper<BigDecimal> BIG_DECIMAL_ROW_MAPPER = new SingleColumnRowMapper<BigDecimal>(BigDecimal.class);
024    public static final RowMapper<byte[]> BYTE_ARRAY_ROW_MAPPER = new SingleColumnRowMapper<byte[]>(byte[].class);
025    public static final RowMapper<Date> DATE_ROW_MAPPER = new SingleColumnRowMapper<Date>(Date.class);
026}
